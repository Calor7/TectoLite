252:     // Helper: Interpolate points along a polyline to a fixed resolution (e.g. 1 degree)
253:     private interpolatePoints(points: Coordinate[], resolution: number): Coordinate[] {
254:         if (points.length < 2) return points;
255:         const result: Coordinate[] = [points[0]];
256: 
257:         for (let i = 0; i < points.length - 1; i++) {
258:             const p1 = points[i];
259:             const p2 = points[i + 1];
260: 
261:             // Spherical distance without import loop? Using class helper or just simple approx?
262:             // SimulationEngine imports from sphericalMath, so we use that.
263:             // But 'distance' wasn't imported. Let's assume linear interpolation on lat/lon for simplicity 
264:             // OR proper slerp if vectors available. 
265:             // Given the typical scale, linear lat/lon interp is "okay" for short segments but bad for long ones.
266:             // Let's use a robust Vector3 slerp/nlerp approach.
267: 
268:             const v1 = latLonToVector(p1);
269:             const v2 = latLonToVector(p2);
270: 
271:             // Dot for angle
272:             // We need 'distance' equivalent. 
273:             // Let's just use a simple approx distance: sqrt(dx^2 + dy^2) is bad for poles.
274:             // Let's reuse rotatePoint logic or just simple linear fraction if segments are small.
275:             // Actually, we should just iterate.
276: 
277:             const dotProd = v1.x * v2.x + v1.y * v2.y + v1.z * v2.z;
278:             const angle = Math.acos(Math.max(-1, Math.min(1, dotProd)));
279:             const distDeg = angle * (180 / Math.PI);
280: 
281:             const numSteps = Math.ceil(distDeg / resolution);
282: 
283:             for (let s = 1; s <= numSteps; s++) {
284:                 const t = s / numSteps;
285:                 // Slerp
286:                 const sinTotal = Math.sin(angle);
287:                 // If angle is too small, linear (nlerp) is fine
288:                 if (sinTotal < 0.001) {
289:                     // Linear mix
290:                     const x = v1.x * (1 - t) + v2.x * t;
291:                     const y = v1.y * (1 - t) + v2.y * t;
292:                     const z = v1.z * (1 - t) + v2.z * t;
293:                     // Normalize
294:                     const mag = Math.sqrt(x * x + y * y + z * z);
295:                     result.push(vectorToLatLon({ x: x / mag, y: y / mag, z: z / mag }));
296:                 } else {
297:                     const ratioA = Math.sin((1 - t) * angle) / sinTotal;
298:                     const ratioB = Math.sin(t * angle) / sinTotal;
299:                     const x = v1.x * ratioA + v2.x * ratioB;
300:                     const y = v1.y * ratioA + v2.y * ratioB;
301:                     const z = v1.z * ratioA + v2.z * ratioB;
302:                     // Already unit length theoretically, but safe to normalize
303:                     result.push(vectorToLatLon({ x, y, z }));
304:                 }
305:             }
306:         }
307:         return result;
308:     }

315:         const newStrips: TectonicPlate[] = [];
316:         const RIFT_GRID_RESOLUTION = 2.0;
317:

319:         const riftPlates = currentPlates.filter(p => p.type === 'rift');
320: 
321:         for (const rift of riftPlates) {
322:             const axisPoly = rift.polygons[0];
323:             if (!axisPoly || !axisPoly.points || axisPoly.points.length < 2) {
324:                 continue;
325:             }
326: 
327:             const axisPoints = this.interpolatePoints(axisPoly.points, RIFT_GRID_RESOLUTION);
328:             const riftBirth = rift.birthTime;
329: 
330:             // Skip if rift hasn't been born yet
331:             if (currentTime <= riftBirth) continue;
332: 
333:             // 2. Find Connected Plates (The Pullers)
334:             // We only care about plates that are explicitly connected to this rift
335:             // AND are "diverging" (moving away).
336:             const connectedPlates = currentPlates.filter(p =>
337:                 (p.connectedRiftIds?.includes(rift.id) || p.connectedRiftId === rift.id) &&
338:                 p.type !== 'rift' &&
339:                 p.type !== 'oceanic' // Only attach to the main lithosphere/continent parent
340:             );
341: 
342: 
343:             for (const plate of connectedPlates) {
344:                 if (plate.riftGenerationMode === 'never') continue;
345:                 if (plate.deathTime !== null && currentTime > plate.deathTime) continue;
346: 
347:                 // Growing strips (_growing suffix) are cleaned up by update() each frame.
348:                 // We only consider permanent strips for backfill scheduling.

354:                             p.linkedToPlateId === plate.id &&

356:                             !p.slabId?.endsWith('_growing') // Ignore growing strips
357:                         )

359:                     return candidates[0];
360:                 };
361: 
362:                 let lastStrip = getLatestPermanentStrip();
363:                 let nextGenerationTime = lastStrip
364:                     ? lastStrip.birthTime + interval
365:                     : (Math.floor(riftBirth / interval) + 1) * interval;
366: 
367:                 // Loop to backfill Permanent Strips
368:                 while (nextGenerationTime <= currentTime) {
369:                     const generationTime = nextGenerationTime;
370:                     nextGenerationTime += interval; // Advance for next loop
371: 
372:                     if (generationTime <= riftBirth) continue;
373:                     const stripId = `${plate.id}_strip_${generationTime}`;

375:                     // Double check existence
376:                     const alreadyExists = currentPlates.some(p => p.slabId === stripId) || newStrips.some(p => p.slabId === stripId);
377:                     if (alreadyExists) continue;

563:     // Helper to move a point forward in time according to a plate's motion history
564:     private applyPlateMotion(point: Coordinate, plate: TectonicPlate, fromTime: number, toTime: number, allPlates: TectonicPlate[]): Coordinate {
565:         let currentP = point;
566:         let time = fromTime;
567: 
568:         // Resolve linked motion (inherit from parent)
569:         let effectivePlate = plate;
570:         if (plate.linkedToPlateId) {
571:             let current = plate;
572:             const visited = new Set<string>();
573:             while (current.linkedToPlateId && !visited.has(current.id)) {
574:                 visited.add(current.id);
575:                 const parent = allPlates.find(p => p.id === current.linkedToPlateId);
576:                 if (!parent) break;
577:                 current = parent;
578:             }
579:             effectivePlate = current;
580:         }
581: 
582:         const keyframes = (effectivePlate.motionKeyframes || []).sort((a, b) => a.time - b.time);
583: 
584:         // If no keyframes, fallback to simple current motion
585:         if (keyframes.length === 0) {
586:             const pole = effectivePlate.motion.eulerPole;
587:             const dt = toTime - fromTime;
588:             if (dt > 1e-6) {
589:                 const angle = pole.rate * dt;
590:                 currentP = rotatePoint(currentP, pole.position, toRad(angle));
591:             }
592:             return currentP;
593:         }
594: 
595:         while (time < toTime) {
596:             let pole = effectivePlate.motion.eulerPole;
597:             let nextBoundary = toTime;
598: 
599:             // Find last keyframe <= time
600:             let activeKFIndex = -1;
601:             for (let i = 0; i < keyframes.length; i++) {
602:                 if (keyframes[i].time <= time) activeKFIndex = i;
603:                 else break;
604:             }
605: 
606:             if (activeKFIndex !== -1) {
607:                 pole = keyframes[activeKFIndex].eulerPole;
608:                 if (activeKFIndex + 1 < keyframes.length) {
609:                     nextBoundary = Math.min(toTime, keyframes[activeKFIndex + 1].time);
610:                 }
611:             } else {
612:                 // Before first keyframe: use first keyframe's pole (assume constant back in time)
613:                 if (keyframes.length > 0) {
614:                     pole = keyframes[0].eulerPole;
615:                     nextBoundary = Math.min(toTime, keyframes[0].time);
616:                 }
617:             }
618: 
619:             const dt = nextBoundary - time;
620:             if (dt > 1e-6) {
621:                 const angle = pole.rate * dt;
